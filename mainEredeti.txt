import sys
import logging
import threading
import asyncio
from flask import Flask, jsonify, render_template
from aiortc.contrib.media import MediaBlackhole

from Bot import Demo

import os
from dotenv import load_dotenv
from Logger import Logger

# Assuming your class is in a file named demo_module.py
# If it's in the same file, just make sure the class is defined above this.

app = Flask(__name__)

# --- Logging Setup ---
handler = logging.StreamHandler(sys.stdout)
formatter = logging.Formatter('[%(asctime)s] %(levelname)s: %(message)s')
handler.setFormatter(formatter)
app.logger.addHandler(handler)
app.logger.setLevel(logging.INFO)

# --- Global State for Background Task ---
demo_instance = None
bg_loop = None
bg_thread = None

#LOAD .ENV
load_dotenv()

#LOGGER
LOGFILE = os.getenv("MAIN_LOG")
log = Logger(LOGFILE)


def start_async_loop(loop):
    """Function to run the asyncio loop indefinitely in a thread."""
    asyncio.set_event_loop(loop)
    loop.run_forever()

#MAIN PAGE
@app.route("/")
def index():
    return render_template("main.html")

#START ROOM
@app.route('/start/<string:room_id>', methods=['POST', 'GET'])
def start_demo(room_id: str):
    global demo_instance, bg_loop, bg_thread

    if demo_instance and not demo_instance._closed:
        log.Logging({"error": "A demo is already running"})
        return jsonify({"error": "A demo is already running"}), 400

    # 1. Setup the Background Loop and Thread if not already running
    if bg_loop is None:
        bg_loop = asyncio.new_event_loop()
        bg_thread = threading.Thread(target=start_async_loop, args=(bg_loop,), daemon=True)
        bg_thread.start()
        app.logger.info("Background asyncio loop started.")
        log.Logging({"info": "Background asyncio loop started."})

    # 2. Instantiate your Demo class
    # Note: Using your class signature: Demo(roomId, LOGFILE, player, recorder, loop)
    demo_instance = Demo(
        roomId=room_id,
        recorder=MediaBlackhole(),
        loop=bg_loop
    )

    # 3. Schedule the 'run' coroutine in the background loop
    asyncio.run_coroutine_threadsafe(demo_instance.run(), bg_loop)

    app.logger.info(f"Demo started for Room: {room_id}")
    log.Logging(f"Demo started for Room: {room_id}")
    return jsonify({"status": "started", "room_id": room_id}), 200


#STOP ROOM
@app.route('/command/stop', methods=['POST', 'GET'])
def stop_bot():
    global demo_instance, bg_loop

    if demo_instance:
        # Schedule the close coroutine and wait for it to complete
        future = asyncio.run_coroutine_threadsafe(demo_instance.close(), bg_loop)

        try:
            # Wait up to 5 seconds for the bot to clean up WebRTC handles
            future.result(timeout=5)
            app.logger.info("Bot cleanup successful.")
            log.Logging("Bot cleanup successful.")
        except Exception as e:
            app.logger.error(f"Error during bot shutdown: {e}")
            log.Logging("Error during bot shutdown.")

        demo_instance = None
        log.Logging({"status": "stopped"})
        return jsonify({"status": "stopped"}), 200

    log.Logging({"error": "No active bot found"})
    return jsonify({"error": "No active bot found"}), 404


@app.route('/status')
def get_status():
    if demo_instance:
        return jsonify({
            "room_id": demo_instance.roomId,
            "closed": getattr(demo_instance, '_closed', False)
        })
    return jsonify({"status": "idle"})


if __name__ == '__main__':
    # use_reloader=False is mandatory when using background threads in Flask Debug mode
    app.run(host='0.0.0.0', port=5000, debug=True, use_reloader=False)
